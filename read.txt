&НаСервере
Функция ИмпортироватьНаСервере()

	ПутьКФайлу = "C:\Temp\Заказчики.json";

	Попытка

		ФайлJSON = Новый Файл(ПутьКФайлу);

		Если НЕ ФайлJSON.Существует() Тогда
			Возврат "Файл не найден: " + ПутьКФайлу;
		КонецЕсли;

		ТекстовыйДокумент = Новый ТекстовыйДокумент;
		ТекстовыйДокумент.Прочитать(ПутьКФайлу, КодировкаТекста.UTF8);

		ТекстJSON = ТекстовыйДокумент.ПолучитьТекст();

		ЧтениеJSON = Новый ЧтениеJSON;
		ЧтениеJSON.УстановитьСтроку(ТекстJSON);

		ДанныеJSON = ПрочитатьJSON(ЧтениеJSON);

		ЧтениеJSON.Закрыть();

		МассивКонтрагентов = ДанныеJSON;
		Количество = 0;

		Для Каждого КонтрагентJSON Из МассивКонтрагентов Цикл

			СсылкаКонтрагента = Справочники.Контрагенты.НайтиПоНаименованию(КонтрагентJSON.name);

			Если СсылкаКонтрагента.Пустая() Тогда
				КонтрагентОбъект = Справочники.Контрагенты.СоздатьЭлемент();
			Иначе
				КонтрагентОбъект = СсылкаКонтрагента.ПолучитьОбъект();
			КонецЕсли;

			КонтрагентОбъект.Наименование = КонтрагентJSON.name;
			КонтрагентОбъект.Номер = КонтрагентJSON.id;
			КонтрагентОбъект.ИНН = КонтрагентJSON.inn;
			КонтрагентОбъект.Адрес = КонтрагентJSON.addres;
			КонтрагентОбъект.Телефон = КонтрагентJSON.phone;
			КонтрагентОбъект.Поставщик = КонтрагентJSON.salesman;
			КонтрагентОбъект.Покупатель = КонтрагентJSON.buyer;

			КонтрагентОбъект.Записать();
			Количество = Количество + 1;

		КонецЦикла;

		Возврат "Импорт завершён. Загружено контрагентов: " + Строка(Количество);

	Исключение

		Возврат "Ошибка при импорте: " + ОписаниеОшибки();

	КонецПопытки;

		Возврат;

	Иначе
		ТекущаяСтрока.Цена = 0;
	КонецЕсли;

Обработка Авторизация — модуль формы
==========================================================================================
Авторизация по справочнику ПользователиЭкзамена, капча из 4 картинок, блокировка после 3 ошибок. В качестве логина используется Наименование пользователя.

&НаКлиенте
Процедура ПриОткрытии(Отказ)

	Авторизован = Ложь;
	НомерКапчи = 0;
	СменитьКапчуНаСледующую();

КонецПроцедуры


&НаКлиенте
Процедура Войти(Команда)

	Если ПустаяСтрока(Логин) Или ПустаяСтрока(Пароль) Тогда
		ПоказатьПредупреждение(, "Введите логин и пароль");
		Возврат;
	КонецЕсли;

	Результат = ПроверитьАвторизациюНаСервере(Логин, Пароль, ПорядокКапчи);

	Если Результат.Успех Тогда

		Сообщить(Результат.Сообщение);
		Авторизован = Истина;

		Если Результат.Роль = "Администратор" Тогда
			ОткрытьФорму("Справочник.ПользователиЭкзамена.ФормаСписка");
		Иначе
			ОткрытьФорму("Документ.ЗаказПокупателя.ФормаСписка");
		КонецЕсли;

		Закрыть();

	Иначе

		СменитьКапчуНаСледующую();
		ПоказатьПредупреждение(, Результат.Сообщение);

	КонецЕсли;

КонецПроцедуры


&НаСервере
Функция ПроверитьАвторизациюНаСервере(Логин, Пароль, ПорядокКапчи)

	Результат = Новый Структура;
	Результат.Вставить("Успех", Ложь);
	Результат.Вставить("Сообщение", "");
	Результат.Вставить("Роль", "");

	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ ПЕРВЫЕ 1
	|  Пользователи.Ссылка КАК Ссылка,
	|  Пользователи.Пароль КАК Пароль,
	|  Пользователи.Роль КАК Роль,
	|  Пользователи.Заблокирован КАК Заблокирован,
	|  Пользователи.ОшибкиВхода КАК ОшибкиВхода
	|ИЗ
	|  Справочник.ПользователиЭкзамена КАК Пользователи
	|ГДЕ
	|  Пользователи.Наименование = &Логин";

	Запрос.УстановитьПараметр("Логин", Логин);

	Выборка = Запрос.Выполнить().Выбрать();

	Если НЕ Выборка.Следующий() Тогда
		Результат.Сообщение = "Неверный логин или пароль";
		Возврат Результат;
	КонецЕсли;

	Если Выборка.Заблокирован Тогда
		Результат.Сообщение = "Пользователь заблокирован. Обратитесь к администратору";
		Возврат Результат;
	КонецЕсли;

	ПользовательОбъект = Выборка.Ссылка.ПолучитьОбъект();

	Если Выборка.Пароль <> Пароль Или ПорядокКапчи <> "1234" Тогда

		ПользовательОбъект.ОшибкиВхода = ПользовательОбъект.ОшибкиВхода + 1;

		Если ПользовательОбъект.ОшибкиВхода >= 3 Тогда

			ПользовательОбъект.Заблокирован = Истина;
			Результат.Сообщение = "Пользователь заблокирован после 3 неверных попыток";

		Иначе

			Если ПорядокКапчи <> "1234" Тогда
				Результат.Сообщение = "Капча собрана неверно. Попытка "
					+ Строка(ПользовательОбъект.ОшибкиВхода) + " из 3";
			Иначе
				Результат.Сообщение = "Неверный логин или пароль. Попытка "
					+ Строка(ПользовательОбъект.ОшибкиВхода) + " из 3";
			КонецЕсли;

		КонецЕсли;

		ПользовательОбъект.Записать();
		Возврат Результат;

	КонецЕсли;

	ПользовательОбъект.ОшибкиВхода = 0;
	ПользовательОбъект.Записать();

	Результат.Успех = Истина;
	Результат.Сообщение = "Вы успешно авторизовались";

	Если Выборка.Роль = Перечисления.РолиПользователей.Администратор Тогда
		Результат.Роль = "Администратор";
	Иначе
		Результат.Роль = "Пользователь";
	КонецЕсли;

	Возврат Результат;

&НаКлиенте

	Если НЕ ЗначениеЗаполнено(Заказ) Тогда
		ПоказатьПредупреждение(, "Выберите заказ покупателя");
		Возврат;
	КонецЕсли;

&НаСервере
Функция СформироватьНаСервере()

	Результат.Очистить();

	ИтогоСырье = 0;
	ИтогоПродукция = 0;

	Запрос = Новый Запрос;

	Запрос.Текст =
	"ВЫБРАТЬ
	|  Товары.Номенклатура.Наименование КАК Продукция,
	|  Товары.Количество КАК КоличествоПродукции,
	|  Товары.Номенклатура.Цена КАК ЦенаПродукции,
	|  Состав.Материал.Наименование КАК Материал,
	|  Состав.Количество КАК КоличествоМатериала,
	|  Состав.Материал.Цена КАК ЦенаМатериала
	|ИЗ
	|  Документ.ЗаказПокупателя.Товары КАК Товары
	|    ВНУТРЕННЕЕ СОЕДИНЕНИЕ Справочник.Спецификации.Состав КАК Состав
	|    ПО Товары.Номенклатура.Наименование = Состав.Ссылка.Продукция.Наименование
	|ГДЕ
	|  Товары.Ссылка = &Заказ
	|
	|УПОРЯДОЧИТЬ ПО
	|  Продукция,
	|  Материал";

	Запрос.УстановитьПараметр("Заказ", Заказ);

	Выборка = Запрос.Выполнить().Выбрать();

	КоличествоСтрок = 0;
	ПредыдущаяПродукция = "";

	Пока Выборка.Следующий() Цикл

		НоваяСтрока = Результат.Добавить();

		КоличествоПродукции = ПолучитьЧисло(Выборка.КоличествоПродукции);
		КоличествоМатериала = ПолучитьЧисло(Выборка.КоличествоМатериала);
		ЦенаМатериала = ПолучитьЧисло(Выборка.ЦенаМатериала);
		ЦенаПродукции = ПолучитьЧисло(Выборка.ЦенаПродукции);

		СтоимостьСырьяСтроки = КоличествоПродукции * КоличествоМатериала * ЦенаМатериала;

		Если Выборка.Продукция <> ПредыдущаяПродукция Тогда

			НоваяСтрока.Продукция = Выборка.Продукция;
			ИтогоПродукция = ИтогоПродукция + КоличествоПродукции * ЦенаПродукции;
			ПредыдущаяПродукция = Выборка.Продукция;

		Иначе

			НоваяСтрока.Продукция = "";

		КонецЕсли;

		НоваяСтрока.Материал = Выборка.Материал;
		НоваяСтрока.КоличествоПродукции = КоличествоПродукции;
		НоваяСтрока.КоличествоМатериала = КоличествоМатериала;
		НоваяСтрока.ЦенаМатериала = ЦенаМатериала;
		НоваяСтрока.Стоимость = СтоимостьСырьяСтроки;

		ИтогоСырье = ИтогоСырье + СтоимостьСырьяСтроки;
		КоличествоСтрок = КоличествоСтрок + 1;

	КонецЦикла;

	Если КоличествоСтрок = 0 Тогда
		ИтогоСырье = 0;
		ИтогоПродукция = 0;
		Возврат "По выбранному заказу нет данных. Проверьте, есть ли спецификация для товаров заказа.";
	КонецЕсли;

	Возврат "Отчёт сформирован. Итого по сырью: " + Строка(ИтогоСырье)
		+ ". Итого по продукции: " + Строка(ИтогоПродукция);
